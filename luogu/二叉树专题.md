# FBI树：
![](二叉树pictrue/Pasted%20image%2020260326165419.png)
```c++
#include <bits/stdc++.h>
using namespace std;
string s;
char get_type(int l, int r) {
    bool has0 = false, has1 = false;

    for(int i = l; i <= r; i ++ ) {
        if(s[i] == '0') has0 = true;
        else has1 = true;
    }

    if(has0 && has1) return 'F';
    if(has0) return 'B';
    return 'I';
}

void dfs(int l, int r) {
    if(l > r) return;

    char type = get_type(l, r);

    if(l != r) {
        int mid = l + r >> 1;

        dfs(l, mid);
        dfs(mid + 1, r);
    }

    cout << type;
}

int main() {
    int n;
    cin >> n;
    cin >> s;

    dfs(0, s.size() - 1);
    return 0;
}
```

# 求先序排列：
![](二叉树pictrue/Pasted%20image%2020260326165529.png)
```c++
#include <bits/stdc++.h>
using namespace std;

string inord, postord;

void build(int il,int ir,int pl,int pr)
{
    if(il>ir) return;

    // 根
    char root = postord[pr];
    cout<<root;

    // 在中序中找根
    int k = il;
    while(inord[k]!=root) k++;

    int left = k - il;

    // 左子树
    build(il, k-1, pl, pl+left-1);

    // 右子树
    build(k+1, ir, pl+left, pr-1);
}

int main()
{
    cin>>inord>>postord;

    build(0,inord.size()-1,0,postord.size()-1);
}
```

# 新二叉树：
![](二叉树pictrue/Pasted%20image%2020260326165540.png)
```c++
#include <bits/stdc++.h>
using namespace std;

char l[30], r[30];
int n;

void dfs(char u) {
    if(u == '*') return;

    cout << u;
    dfs(l[u]);
    dfs(r[u]);
}

int main() {
    cin >> n;
    char root;

    for(int i = 0; i < n; i ++ ) {
        string s;
        cin >> s;

        if(i == 0) root = s[0];

        l[s[0]] = s[1];
        r[s[0]] = s[2];
    }
    dfs(root);

    return 0;
}
```
