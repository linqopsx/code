## ==狡猾的商人== ----$p_{2294}$
带有前缀和及差分思想
```c++
#include <bits/stdc++.h>

using namespace std;

const int MAXN = 100005;

int q[MAXN];

int dist[MAXN];

int find(int x) {

    if (q[x] != x) {

        int parent = q[x];

        q[x] = find(parent);

        dist[x] += dist[parent];

    }

    return q[x];

}

int main() {

    ios::sync_with_stdio(false);

    cin.tie(nullptr);

    int w;

    cin >> w;

    while (w--) {

        int n, m;

        cin >> n >> m;

        bool valid = true;

        for (int i = 0; i <= n; i++) {

            q[i] = i;

            dist[i] = 0;

        }  

        for (int i = 0; i < m; i++) {

            int s, t, v;

            cin >> s >> t >> v;

            s--;

            int root_s = find(s);

            int root_t = find(t);

            if (root_s != root_t) {

                q[root_s] = root_t;

                dist[root_s] = dist[t] - dist[s] - v;

            } else {

                if (dist[t] - dist[s] != v) {

                    valid = false;

                }

            }

        }

        cout << (valid ? "true" : "false") << '\n';

    }

  

    return 0;

  

}
```

## ==银河英雄传说==---- $p_{1196}$
```c++
#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

const int MAXN = 30000;
int parent[MAXN + 1];  // 并查集父节点数组
int dist[MAXN + 1];    // 节点到根节点的距离（位置偏移）
int size[MAXN + 1];    // 每列战舰数量（仅在根节点有效）

// 查找根节点并进行路径压缩，同时更新距离
int find(int x) {
    if (parent[x] != x) {
        int orig_parent = parent[x];
        parent[x] = find(parent[x]);      // 递归压缩路径
        dist[x] += dist[orig_parent];      // 更新到根节点的距离
    }
    return parent[x];
}

int main() {
    // 初始化并查集
    for (int i = 1; i <= MAXN; i++) {
        parent[i] = i;
        dist[i] = 0;
        size[i] = 1;
    }

    int T;
    scanf("%d", &T);
    char op[2];
    int i, j;

    while (T--) {
        scanf("%s %d %d", op, &i, &j);
        if (op[0] == 'M') {
            // 合并操作：将i所在的列接到j所在列的尾部
            int root_i = find(i);
            int root_j = find(j);
            if (root_i != root_j) {
                parent[root_i] = root_j;           // 将root_i挂到root_j下
                dist[root_i] = size[root_j];       // root_i的新位置 = j列的长度
                size[root_j] += size[root_i];      // 更新j列的总长度
            }
        } else {
            // 查询操作
            int root_i = find(i);
            int root_j = find(j);
            if (root_i != root_j) {
                printf("-1\n");
            } else {
                // 计算两艘战舰之间的间隔数（不包括自身）
                int ans = abs(dist[i] - dist[j]) - 1;
                printf("%d\n", ans);
            }
        }
    }
    return 0;
}
```
